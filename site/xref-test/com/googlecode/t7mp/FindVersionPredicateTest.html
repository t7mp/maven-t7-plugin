18  import java.util.Collection;
19  import java.util.List;

21  import org.apache.maven.model.Dependency;
22  import org.junit.Assert;
23  import org.junit.Test;
24  import org.mockito.Mockito;

26  import com.google.common.collect.Collections2;
27  import com.google.common.collect.Lists;
28  import com.googlecode.t7mp.AbstractArtifact;
29  import com.googlecode.t7mp.FindVersionPredicate;
30  import com.googlecode.t7mp.JarArtifact;
31  import com.googlecode.t7mp.PluginLog;
32  
33  public class FindVersionPredicateTest {
34  
35      private final PluginLog log = Mockito.mock(PluginLog.class);
36  
37      @Test
38      public void findVersionPredicateTest() {
39          FindVersionPredicate predicate = new FindVersionPredicate(getLibs(), log);
40          Collection<Dependency> deps = Collections2.filter(getDependencies(), predicate);
41          Assert.assertNotNull(deps);
42          Assert.assertFalse(deps.isEmpty());
43      }
44  
45      private List<Dependency> getDependencies() {
46          List<Dependency> result = Lists.newArrayList();
47          Dependency dep = new Dependency();
48          dep.setGroupId("commons-logging");
49          dep.setArtifactId("commons-logging");
50          dep.setVersion("1.1.1");
51          dep.setClassifier("sources");
52          result.add(dep);
53          return result;
54      }
55  
56      private List<AbstractArtifact> getLibs() {
57          List<AbstractArtifact> result = Lists.newArrayList();
58          JarArtifact jar = new JarArtifact();
59          jar.setArtifactId("commons-logging");
60          jar.setGroupId("commons-logging");
61          jar.setClassifier("sources");
62          jar.setType("jar");
63          result.add(jar);
64          return result;
65      }
